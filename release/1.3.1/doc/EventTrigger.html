new EventTrigger()

Example
const et = new Scene.EventTrigger();
  const scene = new Scene();

  scene.on("call", e => {
      console.log(e.param);
  });
  et.on("call", e => {
      console.log(e.param);
  });
  scene.trigger("call", {param: 1});
  et.trigger("call", {param: 1});
